CREATE TABLE CLIENT
(
    ID NUMBER(10) NOT NULL,   
    NAME VARCHAR2(4000) NOT NULL, 
    EMAIL VARCHAR2(20) NOT NULL,  
    AGE NUMBER(2) NULL, 
    PRIMARY KEY (ID)
);

CREATE TABLE MESSAGE
(
    ID NUMBER(10) NOT NULL,   
    MESSAGETEXT VARCHAR2(4000) NOT NULL,  
    PRIMARY KEY (ID)
);
